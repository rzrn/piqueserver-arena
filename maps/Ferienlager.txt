"""
This is a dedicated building map. All changes are permanent and automatically saved in the “saves/<seed>.vxl” file,
which you can load later, so that you may switch to another map, player a match on it and return to this sandbox.
To load map with the seed N, use `/loadmap Ferienlager#N` command. This way you can manage multiple saves.

In particular, it means that you can save an arbitrary map as “saves/X.vxl” and load it as `/loadmap Ferienlager#X`.
"""

from os.path import dirname
from os import makedirs

from arenalib.maptools import WorldVXL, CTF, dump_on_map_unloaded

name        = "Ferienlager"
team1_name  = "Children"
team1_color = (255, 96, 96)
team2_name  = "Adults"
team2_color = (96, 96, 255)
cap_limit   = 99

extensions = CTF(
    arena_blue_spawn      = (122, 256, 0),
    arena_green_spawn     = (390, 256, 0),
    arena_blue_base       = (128, 256, 0),
    arena_green_base      = (384, 256, 0),
    arena_give_autorefill = True,
    arena_respawn_time    = 3
)

on_map_unloaded = dump_on_map_unloaded

def on_seed_generation(rot_info):
    if seed := rot_info.seed:
        return seed
    else:
        # To avoid wasting space with random seeds in the “saves/” directory.
        return 1

def gen_script(basename, seed):
    filename = "saves/{}.vxl".format(seed)
    makedirs(dirname(filename), exist_ok = True)

    return WorldVXL(filename)
